create database TakeOn;
use TakeOn;

create table usuarios(
ID	INT(10) AUTO_INCREMENT,
NOMBRE		VARCHAR(50),
USUARIO		VARCHAR(50),
PASSWORD	VARCHAR(50),
ESTADO		VARCHAR(50),
ROL		VARCHAR(50),
PRIMARY KEY(ID)
);

create table Productos(
ID	INT(10) AUTO_INCREMENT,
NOMBRE		VARCHAR(50),
CODIGO		VARCHAR(50),
UNIDAD		VARCHAR(50),
CANTIDAD_INI	VARCHAR(50),
PRECIO		VARCHAR(50),
ESTADO		VARCHAR(50),
PRIMARY KEY(ID)
);


create table transaccion(
ID	INT(10) AUTO_INCREMENT,
NROTRAN		VARCHAR(50),
TIPOTRAN	VARCHAR(50),
PRODUCTO	VARCHAR(50),
CANTIDAD	VARCHAR(50),
UNIDAD		VARCHAR(50),
VALUNIT		VARCHAR(50),
VALTOTAL	VARCHAR(50),
CLIENTE		VARCHAR(50),
FECHA		VARCHAR(50),
ESTADO		VARCHAR(50),
PRIMARY KEY(ID)
);

create table clientes(
ID	INT(10) AUTO_INCREMENT,
NOMBRE		VARCHAR(50),
TELEFONO	VARCHAR(50),
DIRECCION	VARCHAR(50),
PRIMARY KEY(ID)
);

create table inventarios(
ID	INT(10) AUTO_INCREMENT,
NOMBRE		VARCHAR(50),
CODIGO		VARCHAR(50),
UNIDAD		VARCHAR(50),
CANTIDAD	VARCHAR(50),
PRECIO		VARCHAR(50),
PRIMARY KEY(ID)
);
